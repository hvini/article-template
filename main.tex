% --- Alteração estilo maketitle ---

\makeatletter
\def\@maketitle{
   \begin{center}
      \let \footnote \thanks
      {\large \bfseries \sffamily \MakeUppercase \imprimirtitulo }\\[2ex]
      {\large \bfseries \sffamily \MakeUppercase \imprimirtituloestrangeiro }\\[4ex]
      {\small
      \lineskip .5em
      \begin{tabular}[t]{c}
         \@author
      \end{tabular}\par}
      {\small \imprimirdata}\\[2ex]
      {\small{Submetido em: 26 jul. de 2018. Aceito em: 20 ago. de 2018.}}\\[4ex]
      \hrule height 1.5pt \vspace{1.2pt}\hrule height 1pt
   \end{center}}
\makeatother
